        \begin{figure}
            \begin{align*}
                &Z_i = Y_i m_i\\
                &Y_i \sim \text{Binomial}(m_i, \pi_i)/m_i\\
                &cauchit(\pi_i) = \eta_i\\
                &\eta_i = \beta_0 +\\
                &+\beta_1 I(T_i="Fictional Novel")
                +\beta_2 I(T_i="Medical")\\
                &+\beta_3 W_i
                +\beta_4 I(\text{Classifier}_i="B")\\
                &+\beta_5 I(\text{Classifier}_i="B") * I(T_i="Fictional Novel")\\
                &+\beta_6 I(\text{Classifier}_i="B") * I(T_i="Medical")
            \end{align*}
            \caption{Article type interaction model. This model is identical to the baseline model with the exception that there are two aditional linear terms to $\eta_i$. $\beta_4$ represents the differencence in cauchit probability of a positive between a text under Calssifier A and Classifier B. $\beta_5$ represents the difference in the value of cauchit probability of a positive between Classifier A and Classifier B, between a fictional novel and an art history novel of the same length. $\beta_6$ is similar to $\beta_5$ except for medical articles instead of fictional novels}
            \label{fig:type_interaction_model}
        \end{figure}
